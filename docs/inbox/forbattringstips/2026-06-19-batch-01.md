# Del 1: Fullständiga Firestore Security Rules

## Målsättning
Säkra alla data enligt RBAC, skydda WORM-data, hobbyvalidera requests och begränsa access.

## Rollmodell
- Admin: All access
- PowerUser: Det mesta förvaltning och analys
- User: Normalt bruk, egna data
- Guest: Minimal, inloggad men begränsad

## Användardata i `/users/{uid}` innehåller roll som en string.

---

## Exempelregel

```rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Hjälpfunktion för rollkontroll
    function hasRole(requiredRoles) {
      return request.auth != null &&
        requiredRoles.hasAny(get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role);
    }

    // User kan läsa och skriva sina egna inkast
    match /inkastUploads/{docId} {
      allow create: if request.auth != null;
      allow read, update, delete: if resource.data.userId == request.auth.uid || hasRole(['Admin', 'PowerUser']);
      
      // Ytterligare validering, tex max storlek, fält etc kan adderas här
      allow update: if request.resource.data.keys().hasAll(['synced']) && hasRole(['Admin','PowerUser']);
    }

    // Append-only collections (WORM)
    match /evolution_ledger/{docId} {
      // Skapa ok, uppdatering förbjudet, läs endast om ägare eller admin
      allow create: if request.auth != null;
      allow update, delete: if false;
      allow read: if resource.data.userId == request.auth.uid || hasRole(['Admin']);
    }

    // Adminområde – fullt access
    match /adminData/{docId} {
      allow read, write: if hasRole(['Admin']);
    }

    // Default deny
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
```

# Del 2: CI/CD pipeline med säkerhetsscan och testkörningar

## Exempel: GitHub Actions workflow i detalj

```yaml
name: CI-CD Pipeline with Security and Testing

on:
  push:
    branches:
      - main
  pull_request:
    branches:
      - main

jobs:
  lint-test-build:
    runs-on: ubuntu-latest
    steps:
      - name: Check out code
        uses: actions/checkout@v3

      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: 20

      - name: Install dependencies
        run: npm ci

      - name: Run ESLint
        run: npm run lint

      - name: Run unit tests
        run: npm test -- --coverage

      - name: Run security scan with Snyk
        uses: snyk/actions/node@master
        env:
          SNYK_TOKEN: ${{ secrets.SNYK_TOKEN }}
        with:
          command: test

      - name: Build project
        run: npm run build

      - name: Deploy to Firebase Hosting & Functions
        uses: FirebaseExtended/action-hosting-deploy@v0
        with:
          repoToken: ${{ secrets.GITHUB_TOKEN }}
          firebaseServiceAccount: ${{ secrets.FIREBASE_SERVICE_ACCOUNT }}
          projectId: your-firebase-project-id
```

### Anmärkningar

- `npm run lint` och `npm test` är scripts definierade i `package.json`.
- Snyk kräver en token; scan är för sårbarhetsanalys.
- Deploy sker först vid lyckad build och tests.
- Lägg till `firebase emulators:start` som extra steg vid PR för lokal testning.

# Del 3: React-komponent och full statehantering för UploadText

### UploadText.tsx – detaljerad version

```tsx
import React, { useState } from 'react';
import { useInkastStore } from '../hooks/useInkastState';
import { v4 as uuidv4 } from 'uuid';

const UploadText: React.FC = () => {
  const [text, setText] = useState('');
  const addUpload = useInkastStore(state => state.addUpload);

  const handleUpload = () => {
    if (!text.trim()) return alert('Fyll i texten');

    const entry = {
      id: uuidv4(),
      type: 'text' as const,
      content: text,
      createdAt: Date.now(),
      synced: false,
    };
    addUpload(entry);
    setText(''); // Reset efter capture
  };

  return (
    <div>
      <textarea
        value={text}
        onChange={e => setText(e.target.value)}
        rows={4}
        placeholder="Skriv ditt inkast här"
        className="border p-2 w-full rounded"
      />
      <button
        onClick={handleUpload}
        className="btn-primary mt-2"
        disabled={!text.trim()}
      >
        Ladda upp textinkast
      </button>
    </div>
  );
};

export default UploadText;
```

# Del 4: Full onboarding wizard med AI-integration

## OnboardingWizard.tsx (React Joyride + AI Chat)

```tsx
import React from 'react';
import Joyride, { Step } from 'react-joyride';
import { useInkastStore } from '../hooks/useInkastState';
import AIChatBubble from './AIChatBubble'; // Extern komponent som interagerar med Vertex AI Chat

const steps: Step[] = [
  { target: '.upload-textarea', content: 'Här kan du skriva ditt första inkast.' },
  { target: '.upload-button', content: 'Tryck här för att ladda upp din text.' },
  { target: '.help-button', content: 'Behöver du hjälp? Klicka här för hjälp från AI.' },
];

const OnboardingWizard = () => {
  const onboardingStep = useInkastStore(state => state.onboardingStep);
  const nextOnboardingStep = useInkastStore(state => state.nextOnboardingStep);

  return (
    <>
      <Joyride
        steps={steps}
        stepIndex={onboardingStep}
        continuous
        showSkipButton
        callback={(data) => {
          if (data.status === 'finished' || data.status === 'skipped') {
            // Logga onboarding-slut
          } else if (data.action === 'next') {
            nextOnboardingStep();
          }
        }}
      />
      <AIChatBubble />
    </>
  );
};

export default OnboardingWizard;
```

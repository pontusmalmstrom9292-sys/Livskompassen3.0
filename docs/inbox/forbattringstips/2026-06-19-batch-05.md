## Del 3: Firestore Security Rules (RBAC & WORM)

```js
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    function hasRole(requiredRoles) {
      return request.auth != null &&
        requiredRoles.hasAny(get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role);
    }

    match /inkastUploads/{docId} {
      allow create: if request.auth != null;
      allow read, update, delete: if resource.data.userId == request.auth.uid || hasRole(['Admin', 'PowerUser']);
      allow update: if request.resource.data.keys().hasAll(['synced']) && hasRole(['Admin','PowerUser']);
    }

    match /evolution_ledger/{docId} {
      allow create: if request.auth != null;
      allow update, delete: if false;
      allow read: if resource.data.userId == request.auth.uid || hasRole(['Admin']);
    }

    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }

    match /{document=**} {
      allow read, write: if false;
    }
  }
}
```

## Del 5: Frontend – React och Zustand Store

### Zustand State Store

```typescript
import create from 'zustand';

export interface InkastEntry {
  id: string;
  type: 'text' | 'image' | 'voice';
  content: string;
  createdAt: number;
  synced: boolean;
}

interface InkastState {
  uploads: InkastEntry[];
  addUpload: (entry: InkastEntry) => void;
  markSynced: (id: string) => void;
  onboardingStep: number;
  nextOnboardingStep: () => void;
}

export const useInkastStore = create<InkastState>((set) => ({
  uploads: [],
  addUpload: (entry) => set((state) => ({ uploads: [...state.uploads, entry] })),
  markSynced: (id) =>
    set((state) => ({
      uploads: state.uploads.map(u => (u.id === id ? { ...u, synced: true } : u)),
    })),
  onboardingStep: 0,
  nextOnboardingStep: () => set(state => ({ onboardingStep: state.onboardingStep + 1 })),
}));
```

### UploadText.tsx-komponent

```tsx
import React, { useState } from 'react';
import { useInkastStore } from '../hooks/useInkastState';
import { v4 as uuidv4 } from 'uuid';

const UploadText: React.FC = () => {
  const [text, setText] = useState('');
  const addUpload = useInkastStore(state => state.addUpload);

  const handleUpload = () => {
    if (!text.trim()) return alert('Fyll i texten');
    const entry = {
      id: uuidv4(),
      type: 'text' as const,
      content: text,
      createdAt: Date.now(),
      synced: false,
    };
    addUpload(entry);
    setText('');
  };

  return    <div>
      <textarea
        value={text}
        onChange={e => setText(e.target.value)}
        rows={4}
        placeholder="Skriv ditt inkast här"
        className="border p-2 w-full rounded"
      />
      <button onClick={handleUpload} className="btn-primary mt-2" disabled={!text.trim()}>
        Ladda upp textinkast
      </button>
    </div>
  );
};

export default UploadText;
```

## Del 6: Backend – Cloud Functions

### submitInkastLite Function

```typescript
import * as functions from 'firebase-functions';
import * as admin from 'firebase-admin';
import { classifyWithVertexAI } from './vertexService';

admin.initializeApp();

export const submitInkastLite = functions.https.onCall(async (data, context) => {
  if (!context.auth) {
    throw new functions.https.HttpsError('unauthenticated', 'User not authenticated');
  }
  
  const { type, content } = data;

  if (!['text', 'image', 'voice'].includes(type)) {
    throw new functions.https.HttpsError('invalid-argument', 'Invalid type');
  }

  const uploadDoc = admin.firestore().collection('inkastUploads').doc();
  await uploadDoc.set({
    userId: context.auth.uid,
    type,
    content,
    createdAt: admin.firestore.FieldValue.serverTimestamp(),
    synced: false,
  });

  const classification = await classifyWithVertexAI(content);
  await uploadDoc.update({ classification });

  return { id: uploadDoc.id, classification };
});
```

## Del 7: Adaptiv Onboarding – React Joyride + AI Chat

### OnboardingWizard.tsx

```tsx
import React from 'react';
import Joyride, { Step } from 'react-joyride';
import { useInkastStore } from '../hooks/useInkastState';
import AIChatBubble from './AIChatBubble';

const steps: Step[] = [
  { target: '.upload-textarea', content: 'Här kan du skriva ditt första inkast.' },
  { target: '.upload-button', content: 'Tryck här för att ladda upp din text.' },
  { target: '.help-button', content: 'Behöver du hjälp? Klicka här för hjälp från AI.' },
];

const OnboardingWizard = () => {
  const onboardingStep = useInkastStore(state => state.onboardingStep);
  const nextOnboardingStep = useInkastStore(state => state.nextOnboardingStep);

  return (
    <>
      <Joyride
        steps={steps}
        stepIndex={onboardingStep}
        continuous
        showSkipButton
        callback={(data) => {
          if (data.status === 'finished' || data.status === 'skipped') {
            // Logga onboarding-slut
          } else if (data.action === 'next') {
            nextOnboardingStep();
          }
        }}
      />
      <AIChatBubble />
    </>
  );
};

export default OnboardingWizard;
```
